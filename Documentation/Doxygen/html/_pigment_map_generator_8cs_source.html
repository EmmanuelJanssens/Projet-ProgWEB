398  //Add the height info to the pigment map's alpha channel
399  private Texture2D AddHeightmapToAlpha(Texture2D pigmentMap, Texture2D inputHeightmap = null)
400  {
401 
402  int spatmapIndex = ((int)heightmapChannel >= 5) ? 1 : 0;
403  int channelIndex = (spatmapIndex > 0) ? (int)heightmapChannel - 4 : (int)heightmapChannel;
404 
405  //Debug.Log("Splatmap index: " + spatmapIndex);
406  //Debug.Log("Channel index: " + channelIndex);
407 
408  heightmap = new Texture2D(pigmentmapSize, pigmentmapSize, TextureFormat.RGB24, false);
409 
410  if (workflow == TerrainUVUtil.Workflow.Terrain)
411  {
412  Texture2D splatmap = terrain.terrainData.alphamapTextures[spatmapIndex];
413 
414  //Use the selected channel from the splatmap to create a heightmap
415  for (int x = 0; x < pigmentmapSize; x++)
416  {
417  for (int y = 0; y < pigmentmapSize; y++)
418  {
419  Color splatmapPixel = splatmap.GetPixel(x, y);
420 
421  switch (channelIndex)
422  {
423  //Red
424  case 1:
425  heightmap.SetPixel(x, y, Color.red * splatmapPixel.r);
426  break;
427  //Green
428  case 2:
429  heightmap.SetPixel(x, y, Color.red * splatmapPixel.g);
430  break;
431  //Blue
432  case 3:
433  heightmap.SetPixel(x, y, Color.red * splatmapPixel.b);
434  break;
435  //Alpha
436  case 4:
437  heightmap.SetPixel(x, y, Color.red * splatmapPixel.a);
438  break;
439  }
440  }
441 
442  heightmap.Apply();
443  }
444  }
445  else if (workflow == TerrainUVUtil.Workflow.Mesh)
446  {
447  //If the input heightmap is of a lower/higher resolution, rescale it
448  if (inputHeightmap.height != pigmentmapSize)
449  {
450  inputHeightmap = ScaleTexture(inputHeightmap, pigmentmapSize, pigmentmapSize);
451  }
452 
453  for (int x = 0; x < pigmentmapSize; x++)
454  {
455  for (int y = 0; y < pigmentmapSize; y++)
456  {
457  Color heightmapPixel = inputHeightmap.GetPixel(x, y);
458 
459  heightmap.SetPixel(x, y, Color.red * heightmapPixel.r);
460  }
461 
462  heightmap.Apply();
463  }
464  }
465 
466  //Create a new pigment map texture
467  if (newPigmentMap)
468  {
469  DestroyImmediate(newPigmentMap);
470  }
471  newPigmentMap = new Texture2D(pigmentmapSize, pigmentmapSize, TextureFormat.ARGB32, false);
472 
473  //Store the heightmap in the alpha channel
474  for (int x = 0; x < pigmentmapSize; x++)
475  {
476  for (int y = 0; y < pigmentmapSize; y++)
477  {
478  float heightPixel = heightmap.GetPixel(x, y).r;
479  Color pigmentmapPixel = pigmentMap.GetPixel(x, y);
480 
481  newPigmentMap.SetPixel(x, y, new Color(pigmentmapPixel.r, pigmentmapPixel.g, pigmentmapPixel.b, heightPixel));
482  }
483  }
484  newPigmentMap.Apply();
485 
486  return newPigmentMap;
487  }

523  //Rescale function by user petersvp
524  public static Texture2D ScaleTexture(Texture2D src, int width, int height)
525  {
526  Rect texRect = new Rect(0, 0, width, height);
527 
528  //We need the source texture in VRAM because we render with it
529  src.filterMode = FilterMode.Trilinear;
530  src.Apply(true);
531 
532  RenderTexture rt = new RenderTexture(width, height, 32);
533 
534  Graphics.SetRenderTarget(rt);
535 
536  //Setup 2D matrix in range 0..1, so nobody needs to care about sized
537  GL.LoadPixelMatrix(0, 1, 1, 0);
538 
539  //Then clear & draw the texture to fill the entire RTT.
540  GL.Clear(true, true, new Color(0, 0, 0, 0));
541  Graphics.DrawTexture(new Rect(0, 0, 1, 1), src);
542 
543  //Get rendered data back to a new texture
544  Texture2D result = new Texture2D(width, height, TextureFormat.ARGB32, true);
545  result.Resize(width, height);
546  result.ReadPixels(texRect, 0, 0, true);
547  result.Apply();
548  return result;
549  }
